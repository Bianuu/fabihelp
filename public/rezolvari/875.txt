#include <iostream>
#include <stack>
#include <string>

using namespace std;

int main() {

    int n;
    cin >> n;

    stack<int> s;
    string op;
    int x;

    for (int i = 0; i < n; ++i) {

        cin >> op;

        if (op == "push") {
            cin >> x;
            s.push(x);
        }
        else if (op == "pop") {
            if (!s.empty())
                s.pop();
        }
        else if (op == "top") {
            if (!s.empty())
                cout << s.top() << '\n';
        }
    }

    return 0;
}
